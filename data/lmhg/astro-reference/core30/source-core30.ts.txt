const categoryModules = import.meta.glob('./copy/categories/*.json', { eager: true, import: 'default' })
const serviceModules = import.meta.glob('./copy/services/*.json', { eager: true, import: 'default' })
const deliveryModeModules = import.meta.glob('./copy/care-settings/*.json', { eager: true, import: 'default' })
const serviceAreaModules = import.meta.glob('./copy/service-areas/*.json', { eager: true, import: 'default' })

function modulesToSlugMap<T>(
  modules: Record<string, unknown>,
  schema: z.ZodType<T>,
) {
  return Object.fromEntries(
    Object.entries(modules).map(([, value]) => {
      const parsed = schema.parse(value)
      // @ts-expect-error slug exists by schema contract
      return [parsed.slug, parsed]
    }),
  ) as Record<string, T>
}

const categoryCopyBySlug = Object.fromEntries(
	Object.entries(modulesToSlugMap(categoryModules, categorySchema)).map(([slug, value]) => [
		slug,
		withNocoBasePageBlocks(`/${slug}/`, `src/data/copy/categories/${slug}.json`, value),
	]),
) as Record<string, Core30Category>
const serviceCopyBySlug = Object.fromEntries(
	Object.entries(modulesToSlugMap(serviceModules, serviceSchema)).map(([slug, value]) => [
		slug,
		withNocoBasePageBlocks(`/${slug}/`, `src/data/copy/services/${slug}.json`, value),
	]),
) as Record<string, Core30Service>
const deliveryModeBySlug = Object.fromEntries(
	Object.entries(modulesToSlugMap(deliveryModeModules, deliveryModeSchema)).map(([slug, value]) => [
		slug,
		withNocoBasePageBlocks(`/locations/${slug}/`, `src/data/copy/care-settings/${slug}.json`, value),
	]),
) as Record<string, DeliveryMode>
const serviceAreaBySlug = Object.fromEntries(
	Object.entries(modulesToSlugMap(serviceAreaModules, serviceAreaSchema)).map(([slug, value]) => [
		slug,
		withNocoBasePageBlocks(`/${slug}/`, `src/data/copy/service-areas/${slug}.json`, value),
	]),
) as Record<string, ServiceAreaPage>

const categoryOrder = [
  'individual-counseling',
  'child-counseling',
  'family-therapy',
  'couples-counseling',
  'court-ordered',
  'community-based-services',
  'group-therapy',
  'trauma-therapy',
] as const

const serviceSlugsByCategory: Record<(typeof categoryOrder)[number], string[]> = {
  'individual-counseling': ['adult-counseling', 'anxiety-depression-therapy'],
  'child-counseling': ['adolescent-counseling', 'play-therapy', 'child-behavioral-intervention'],
  'family-therapy': ['attachment-therapy', 'parenting-support'],
  'couples-counseling': ['relationship-counseling', 'couples-conflict-resolution'],
  'court-ordered': ['family-reunification', 'co-parenting'],
  'community-based-services': ['case-management', 'community-support', 'therapy-in-your-home'],
  'group-therapy': [],
  'trauma-therapy': [],
}

const deliveryModeOrder = ['in-person', 'community', 'in-home', 'online', 'school'] as const
const serviceAreaOrder = ['louisville-ky', 'jefferson-county-ky', 'bullitt-county-ky', 'oldham-county-ky'] as const

export const deliveryModes: DeliveryMode[] = deliveryModeOrder.map((slug) => {
  const mode = deliveryModeBySlug[slug]
  if (!mode) throw new Error(`Missing care setting copy for ${slug}`)
  return mode
})
export const deliveryModeMap = deliveryModeBySlug

export const serviceAreaPages: ServiceAreaPage[] = serviceAreaOrder.map((slug) => {
  const area = serviceAreaBySlug[slug]
  if (!area) throw new Error(`Missing service-area copy for ${slug}`)
  return area
})
export const serviceAreaPageMap = serviceAreaBySlug

export const core30Categories: Core30Category[] = categoryOrder.map((slug) => {
  const category = categoryCopyBySlug[slug]
  if (!category) throw new Error(`Missing Core30 category copy for ${slug}`)
  const services = serviceSlugsByCategory[slug].map((serviceSlug) => {
    const service = serviceCopyBySlug[serviceSlug]
    if (!service) throw new Error(`Missing Core30 service copy for ${serviceSlug}`)
    return service
  })
  return { ...category, services }
})

export const core30CategoryMap = Object.fromEntries(
  core30Categories.map((category) => [category.slug, category]),
) as Record<string, Core30Category>

export const core30Services = core30Categories.flatMap((category) =>
  category.services.map((service) => ({
    ...service,
    categorySlug: category.slug,
    categoryTitle: category.title,
    categoryHeroTitle: category.heroTitle,
  })),
)

export const core30ServiceMap = Object.fromEntries(
  core30Services.map((service) => [service.slug, service]),
) as Record<
  string,
  Core30Service & {
    categorySlug: string
    categoryTitle: string
    categoryHeroTitle: string
  }
>
